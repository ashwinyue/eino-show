```
│  ┌─────────────────────────────────────────────────────────┐    │
```

### 2. 在 Agent 中发送事件

```go
// internal/pkg/agent/chat/agent.go

func (a *Agent) Generate(ctx context.Context, eventBus *event.EventBus, messages []*schema.Message, opts ...model.Option) (*schema.Message, error) {
    // 发送思考事件
    if eventBus != nil {
        _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentThought, event.AgentThoughtData{
            Content:   "Thinking...",
            Iteration: 0,
            Done:      true,
        }))
    }

    // 执行业务逻辑...
    msg, err := a.chatModel.Generate(ctx, messages, opts...)
    if err != nil {
        // 发送错误事件
        if eventBus != nil {
            _ = eventBus.Emit(ctx, event.NewEvent(event.EventError, event.ErrorData{
                Error: err.Error(),
                Stage: "chat_generation",
            }))
        }
        return nil, err
    }

    // 发送最终答案事件
    if eventBus != nil {
        _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentFinalAnswer, event.AgentFinalAnswerData{
            Content: msg.Content,
            Done:    true,
        }))

        // 发送完成事件
        _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentComplete, event.AgentCompleteData{
            TotalSteps:  1,
            FinalAnswer: msg.Content,
        }))
    }

    return msg, nil
}
```

## 在 ReactAgent 中集成

### 核心集成点

ReactAgent 的执行循环是发送事件的最佳位置：

```go
// internal/agent/react/agent.go

import (
    "github.com/ashwinyue/eino-show/internal/pkg/event"
)

// Generate 执行 Agent 生成.
func (a *Agent) Generate(ctx context.Context, eventBus *event.EventBus, messages []*schema.Message) (*schema.Message, error) {
    // 如果没有 EventBus，直接执行（兼容旧代码）
    if eventBus == nil {
        return a.generateWithoutEvents(ctx, messages)
    }

    // 执行循环
    for i := 0; i < a.cfg.MaxIterations; i++ {
        // 1. 发送思考事件
        _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentThought, event.AgentThoughtData{
            Content:   thought,
            Iteration: i,
            Done:      true,
        }))

        // 2. 发送工具调用事件
        for _, toolCall := range toolCalls {
            toolCallID := generateToolCallID()

            _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentToolCall, event.AgentToolCallData{
                ToolCallID: toolCallID,
                ToolName:   toolCall.Function.Name,
                Arguments:  parseArguments(toolCall.Function.Arguments),
                Iteration:  i,
            }))

            // 3. 执行工具
            startTime := time.Now()
            result, err := a.executeTool(ctx, toolCall)
            duration := time.Since(startTime).Milliseconds()

            // 4. 发送工具结果事件
            _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentToolResult, event.AgentToolResultData{
                ToolCallID: toolCallID,
                ToolName:   toolCall.Function.Name,
                Output:     result.Output,
                Error:      getErrorString(err),
                Success:    err == nil,
                Duration:   duration,
                Iteration:  i,
                Data:       result.Data, // 结构化数据 (如 display_type)
            }))
        }

        // 5. 检查是否完成
        if isComplete {
            break
        }

        // 6. 发送反思事件 (可选)
        if reflection != "" {
            _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentReflection, event.AgentReflectionData{
                Content:   reflection,
                Iteration: i,
                Done:      true,
            }))
        }
    }

    // 7. 发送最终答案事件 (流式)
    for _, chunk := range finalAnswerChunks {
        _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentFinalAnswer, event.AgentFinalAnswerData{
            Content: chunk,
            Done:    false,
        }))
    }
    _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentFinalAnswer, event.AgentFinalAnswerData{
        Content: "",
        Done:    true,
    }))

    // 8. 发送完成事件
    _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentComplete, event.AgentCompleteData{
        TotalSteps:      iteration,
        FinalAnswer:     finalAnswer,
        KnowledgeRefs:   knowledgeRefs,
        TotalDurationMs: time.Since(start).Milliseconds(),
    }))

    return finalMessage, nil
}
```

### 工具调用事件示例

```go
// 知识库搜索工具事件
func (t *KnowledgeSearchTool) Run(ctx context.Context, eventBus *event.EventBus, args map[string]any) (*ToolResult, error) {
    query := args["query"].(string)
    topK := int(args["top_k"].(float64))

    // 发送工具调用事件
    toolCallID := uuid.New().String()
    if eventBus != nil {
        _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentToolCall, event.AgentToolCallData{
            ToolCallID: toolCallID,
            ToolName:   "KnowledgeSearch",
            Arguments:  args,
        }))
    }

    // 执行搜索
    startTime := time.Now()
    results, err := t.store.Search(ctx, query, topK)
    duration := time.Since(startTime).Milliseconds()

    // 构建结构化数据
    data := map[string]any{
        "display_type": "knowledge_chunks",
        "formatted":    formatResults(results),
    }

    // 发送工具结果事件
    if eventBus != nil {
        _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentToolResult, event.AgentToolResultData{
            ToolCallID: toolCallID,
            ToolName:   "KnowledgeSearch",
            Output:     summarizeResults(results),
            Success:    err == nil,
            Duration:   duration,
            Data:       data,
        }))
    }

    return &ToolResult{Data: results}, err
}
```

```go
// internal/pkg/event/stream_handler.go

func (h *AgentStreamHandler) handleProgress(ctx context.Context, evt Event) error {
    data, ok := evt.Data.(AgentProgressData)
    if !ok {
        return nil
    }

    resp := StreamResponse{
        ID:           evt.ID,
        ResponseType: ResponseProgress,
        Content:      data.Message,
        SessionID:    h.sessionID,
        Data: map[string]interface{}{
            "current_step": data.CurrentStep,
            "total_steps":  data.TotalSteps,
            "percent":      data.Percent,
        },
    }

    return h.writeStreamResponse(ctx, resp)
}

// 在 Subscribe 中注册
func (h *AgentStreamHandler) Subscribe() {
    // 现有订阅...
    h.eventBus.On(EventAgentProgress, h.handleProgress)
}
```

## 前端对接

### SSE 连接

```typescript
// frontend/src/api/chat/stream.ts

export async function* chatStream(
  sessionId: string,
  question: string
): AsyncGenerator<StreamResponse> {
  const response = await fetch(`/api/v1/sessions/${sessionId}/qa`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ question }),
  });

  const reader = response.body!.getReader();
  const decoder = new TextDecoder();
  let buffer = '';

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split('\n');
    buffer = lines.pop() || '';

    for (const line of lines) {
      if (line.startsWith('data: ')) {
        const data = JSON.parse(line.slice(6)) as StreamResponse;
        yield data;
      }
    }
  }
}

// 使用示例
for await (const event of chatStream(sessionId, question)) {
  switch (event.response_type) {
    case 'thinking':
      updateThinkingContent(event.id, event.content);
      break;
    case 'tool_call':
      addToolCall(event.data.tool_name);
      break;
    case 'tool_result':
      updateToolResult(event.data.tool_call_id, event.data);
      break;
    case 'answer':
      appendAnswerContent(event.id, event.content);
      break;
    case 'complete':
      markComplete();
      break;
  }
}
```

### 事件累积处理

```typescript
// 前端需要根据 event.id 累积流式内容

const eventAccumulators = new Map<string, string>();

function processStreamEvent(event: StreamResponse) {
  if (event.response_type === 'thinking' || event.response_type === 'answer') {
    // 累积内容
    const current = eventAccumulators.get(event.id) || '';
    const updated = current + event.content;
    eventAccumulators.set(event.id, updated);

    // 更新 UI
    updateDisplay(event.id, updated);

    // 如果完成，清理累积器
    if (event.done) {
      eventAccumulators.delete(event.id);
    }
  }
}
```

## 最佳实践

### 1. 事件 ID 生成

```go
// 使用 UUID 确保唯一性
eventID := uuid.New().String()

// 或使用工具调用 ID 作为事件 ID 前缀
toolCallID := "tool-" + uuid.New().String()
```

### 2. 流式事件分块

```go
// 对于长内容，分块发送
chunks := splitContent(content, 256) // 每块 256 字符

for i, chunk := range chunks {
    _ = eventBus.Emit(ctx, event.NewEvent(event.EventAgentFinalAnswer, event.AgentFinalAnswerData{
        Content: chunk,
        Done:    i == len(chunks)-1, // 最后一块标记为完成
    }))
}
```

### 3. 错误处理

```go
// 始终在错误时发送事件
if err != nil {
    _ = eventBus.Emit(ctx, event.NewEvent(event.EventError, event.ErrorData{
        Error:     err.Error(),
        ErrorCode: "TOOL_EXECUTION_FAILED",
        Stage:     "tool_call",
    }))
    return nil, err
}
```

### 4. 性能考虑

```go
// 使用异步模式避免阻塞
eventBus := event.NewAsyncEventBus()
```
